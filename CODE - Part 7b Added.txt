# ============================================================
# 2) TEXT CLEANING
# ============================================================

# ------------------------------------------------------------
# Standardised preprocessing function applied to all text
# Reduces formatting noise while preserving semantic content
# ------------------------------------------------------------
def clean_text(text):

    # --------------------------------------------------------
    # Convert to lowercase for vocabulary consistency
    # --------------------------------------------------------
    text = text.lower()

    # --------------------------------------------------------
    # Remove URLs - not informative for classification
    # --------------------------------------------------------
    text = re.sub(r"http\S+", " ", text)

    # --------------------------------------------------------
    # Replace digits with X
    # Prevents phone number memorisation (privacy + leakage)
    # --------------------------------------------------------
    text = re.sub(r"\d", "X", text)

    # --------------------------------------------------------
    # Remove repeated punctuation (e.g. !!!, ???)
    # --------------------------------------------------------
    text = re.sub(r"[!?.]{2,}", " ", text)

    # --------------------------------------------------------
    # Normalise whitespace to single spaces
    # --------------------------------------------------------
    text = re.sub(r"\s+", " ", text).strip()

    # --------------------------------------------------------
    # Return cleaned text
    # --------------------------------------------------------
    return text

# ------------------------------------------------------------
# Apply cleaning function to entire dataset
# ------------------------------------------------------------

# 6) TRAIN AUGMENTATION
# ============================================================

# ------------------------------------------------------------
# Data augmentation improves model robustness
# Prevents overfitting to exact training phrase patterns
# Applied ONLY to training data - never to val or test
# ------------------------------------------------------------
def augment_text(text):

    # --------------------------------------------------------
    # Split text into individual words
    # --------------------------------------------------------
    words = text.split()

    # --------------------------------------------------------
    # Random word dropout: 10% chance per word
    # Simulates missing or truncated advertisement text
    # --------------------------------------------------------
    words = [
        w for w in words
        if np.random.rand() > 0.1
    ]

    # --------------------------------------------------------
    # Random adjacent word swap: 30% chance
    # Simulates rephrased or reordered advertisement content
    # --------------------------------------------------------
    if len(words) > 4 and np.random.rand() < 0.3:

        # random index for swap
        i = np.random.randint(
            0,
            len(words)-1
        )

        # swap neighbouring words
        words[i], words[i+1] = (
            words[i+1],
            words[i]
        )

    # --------------------------------------------------------
    # Return augmented text
    # --------------------------------------------------------
    return " ".join(words)

# ------------------------------------------------------------
# STEP 7b-1: CHARACTER N-GRAM TF-IDF FEATURES
# analyzer="char_wb" extracts character sequences within
# word boundaries - less prone to memorising full phrases
# ngram_range=(3,6) captures subword and morphological
# patterns common in coded trafficking language
# min_df=3 removes noise from very rare character sequences
# max_features=15000 limits to most informative sequences
# sublinear_tf=True applies log scaling to term frequencies
# strip_accents normalises accented characters
# ------------------------------------------------------------
baseline_vectorizer = TfidfVectorizer(

    # character n-grams within word boundaries
    analyzer="char_wb",

    # 3 to 6 character sequences
    ngram_range=(3, 6),

    # larger vocabulary needed for character n-grams
    max_features=15000,

    # log-scaled TF reduces dominance of frequent patterns
    sublinear_tf=True,

    # normalise accented characters
    strip_accents="unicode",

    # ignore character sequences appearing fewer than 3 times
    min_df=3
)

# ------------------------------------------------------------

# ------------------------------------------------------------

# ------------------------------------------------------------
# STEP 7b-2: BASELINE 1 - TF-IDF + LOGISTIC REGRESSION
# class_weight="balanced" mirrors DistilBERT weighted loss
# C controls inverse regularisation strength:
#   smaller C = stronger regularisation = less overfitting
# Conservative C range [0.001, 0.01, 0.1, 1.0] used

# Test set never used during hyperparameter selection
# ------------------------------------------------------------

# ------------------------------------------------------------

# ------------------------------------------------------------
# Grid search over conservative C range
# Validation set used - test set never seen during tuning
# ------------------------------------------------------------

    # --------------------------------------------------------
    # Train temporary LR classifier with current C value
    # --------------------------------------------------------

        # balance class weights for minority class
        class_weight="balanced",

        # sufficient iterations for convergence
        max_iter=1000,

        # reproducibility
        random_state=SEED,

        # current regularisation value
        C=C
    )

    # --------------------------------------------------------
    # Fit temporary classifier on training data
    # --------------------------------------------------------
    lr_tmp.fit(X_bl_train, y_bl_train)

    # --------------------------------------------------------
    # Predict on validation set for tuning only
    # --------------------------------------------------------
    val_pred = lr_tmp.predict(X_bl_valid)

    # --------------------------------------------------------
    # Compute macro F1 on validation set
    # Macro F1 used as primary metric - consistent with
    # DistilBERT evaluation protocol in the paper
    # --------------------------------------------------------
    val_f1 = f1_score(
        y_bl_valid,
        val_pred,
        average="macro"
    )

    # --------------------------------------------------------
    # Print validation F1 for this C value
    # --------------------------------------------------------
    print(f"  C={C:.3f}  val macro F1={val_f1:.4f}")

    # --------------------------------------------------------
    # Update best C if current C gives better validation F1
    # --------------------------------------------------------
    if val_f1 > best_lr_f1:
        best_lr_f1 = val_f1
        best_lr_C  = C

# ------------------------------------------------------------
# Print selected best C value from validation tuning
# ------------------------------------------------------------

# ------------------------------------------------------------

# Uses best C selected from validation tuning above
# ------------------------------------------------------------

    class_weight="balanced",
    max_iter=1000,
    random_state=SEED,

# ------------------------------------------------------------
# Custom PyTorch dataset wraps tokenised advertisements
# Handles tokenisation, padding, and label conversion
# ------------------------------------------------------------
class AdsDataset(Dataset):

    # --------------------------------------------------------
    # Constructor: store texts and labels as lists
    # --------------------------------------------------------
    def __init__(self, texts, labels):
        self.texts  = texts.tolist()
        self.labels = labels.tolist()

    # --------------------------------------------------------
    # Returns total number of samples in the dataset
    # --------------------------------------------------------
    def __len__(self):
        return len(self.texts)

    # --------------------------------------------------------
    # Returns a single tokenised and labelled sample
    # Called by DataLoader during training and evaluation
    # --------------------------------------------------------
    def __getitem__(self, idx):

        # ----------------------------------------------------
        # Tokenise text with padding and truncation
        # Returns input_ids and attention_mask tensors
        # ----------------------------------------------------
        encoding = tokenizer(

            self.texts[idx],

            # truncate sequences longer than MAX_LEN
            truncation=True,

            # pad sequences shorter than MAX_LEN
            padding="max_length",

            # fixed sequence length for batch consistency
            max_length=MAX_LEN,

            # return PyTorch tensors
            return_tensors="pt"
        )

        # ----------------------------------------------------
        # Remove batch dimension added by tokenizer
        # DataLoader adds it back during collation
        # ----------------------------------------------------
        item = {
            k: v.squeeze(0)
            for k, v in encoding.items()
        }

        # ----------------------------------------------------
        # Add integer label tensor for loss computation
        # ----------------------------------------------------
        item["labels"] = torch.tensor(
            self.labels[idx],
            dtype=torch.long
        )

        # ----------------------------------------------------
        # Return processed sample dictionary
        # ----------------------------------------------------
        return item

# 11) PARTIAL UNFREEZING
# ============================================================

# ------------------------------------------------------------
# Only fine-tune the final transformer layers and classifier
# Freezing early layers preserves general language knowledge
# Reduces overfitting risk on small dataset
# layers 4 and 5 are the final two transformer blocks
# pre_classifier and classifier are the task-specific head
# ------------------------------------------------------------
unfreeze_layers = {
    "layer.4",
    "layer.5",
    "classifier",
    "pre_classifier"

        for layer_name in unfreeze_layers

    p.numel()
    for p in model.parameters()

    p.numel()
    for p in model.parameters()
)

# ------------------------------------------------------------
# Named tuple for structured evaluation output
# Improves readability when accessing evaluation results
# ------------------------------------------------------------
EvalResult = namedtuple(

    "EvalResult",

    [
        "loss",    # average validation/test loss
        "acc",     # overall accuracy
        "f1",      # macro F1-score
        "labels",  # true labels list
        "preds"    # predicted labels list
    ]
)

# ------------------------------------------------------------
# Evaluation function used for validation and test sets
# threshold parameter allows recall/precision trade-off
# Default threshold=0.35 prioritises recall for missed cases
# ------------------------------------------------------------
def evaluate(loader, threshold=0.35):

    # --------------------------------------------------------
    # Switch model to evaluation mode
    # Disables dropout and batch normalisation
    # --------------------------------------------------------
    model.eval()

    # --------------------------------------------------------
    # Storage for predictions and labels across all batches
    # --------------------------------------------------------
    all_preds  = []
    all_labels = []
    total_loss = 0

    # --------------------------------------------------------
    # Disable gradient computation for efficiency
    # --------------------------------------------------------
    with torch.no_grad():

        # ----------------------------------------------------
        # Iterate through all batches in the loader
        # ----------------------------------------------------
        for batch in loader:

            # ------------------------------------------------
            # Move all tensors to the selected device
            # ------------------------------------------------
            batch = {
                k: v.to(device)
                for k, v in batch.items()
            }

            # ------------------------------------------------
            # Forward pass through DistilBERT
            # ------------------------------------------------
            outputs = model(**batch)

            # ------------------------------------------------
            # Extract raw prediction scores (logits)
            # ------------------------------------------------
            logits = outputs.logits

            # ------------------------------------------------
            # Compute weighted loss for monitoring
            # ------------------------------------------------
            loss = loss_fn(
                logits,
                batch["labels"]
            )

            # ------------------------------------------------
            # Accumulate batch loss
            # ------------------------------------------------
            total_loss += loss.item()

            # ------------------------------------------------
            # Convert logits to probabilities using softmax
            # ------------------------------------------------
            probs = torch.softmax(
                logits,
                dim=1
            )

            # ------------------------------------------------
            # Apply threshold to trafficking probability
            # Scores >= threshold classified as trafficking
            # ------------------------------------------------
            preds = (
                probs[:, 1] >= threshold
            ).long()

            # ------------------------------------------------
            # Store batch predictions and true labels
            # ------------------------------------------------
            all_preds.extend(
                preds.cpu().numpy()
            )

            all_labels.extend(
                batch["labels"].cpu().numpy()
            )

    # --------------------------------------------------------
    # Compute accuracy over full evaluation set
    # --------------------------------------------------------
    acc = accuracy_score(
        all_labels,
        all_preds
    )

    # --------------------------------------------------------
    # Compute macro F1 over full evaluation set
    # Primary metric for trafficking detection task
    # --------------------------------------------------------
    f1 = f1_score(
        all_labels,
        all_preds,
        average="macro"
    )

    # --------------------------------------------------------
    # Return structured evaluation result
    # --------------------------------------------------------
    return EvalResult(

        loss=total_loss / len(loader),
        acc=acc,
        f1=f1,
        labels=all_labels,
        preds=all_preds
    )

# 14) TRAINING LOOP
# ============================================================

# ------------------------------------------------------------
# Track best validation loss for model checkpointing
# ------------------------------------------------------------
best_val_loss = float("inf")

# ------------------------------------------------------------
# Early stopping patience: stop if no improvement for 3 epochs
# ------------------------------------------------------------
patience = 3

# ------------------------------------------------------------
# Early stopping counter
# ------------------------------------------------------------
counter = 0

# ------------------------------------------------------------
# Training loop over epochs
# ------------------------------------------------------------
for epoch in range(epochs):

    # --------------------------------------------------------
    # Switch model to training mode
    # Enables dropout and gradient computation
    # --------------------------------------------------------
    model.train()

    # --------------------------------------------------------
    # Reset epoch loss accumulator
    # --------------------------------------------------------
    total_loss = 0

    # --------------------------------------------------------
    # Iterate through all training batches
    # --------------------------------------------------------
    for batch in train_loader:

        # ----------------------------------------------------
        # Move batch tensors to device
        # ----------------------------------------------------
        batch = {
            k: v.to(device)
            for k, v in batch.items()
        }

        # ----------------------------------------------------
        # Forward pass through model
        # ----------------------------------------------------
        outputs = model(**batch)

        # ----------------------------------------------------
        # Extract logits from model output
        # ----------------------------------------------------
        logits = outputs.logits

        # ----------------------------------------------------
        # Compute weighted cross-entropy loss
        # ----------------------------------------------------
        loss = loss_fn(
            logits,
            batch["labels"]
        )

        # ----------------------------------------------------
        # Reset gradients before backward pass
        # ----------------------------------------------------
        optimizer.zero_grad()

        # ----------------------------------------------------
        # Backward pass: compute gradients
        # ----------------------------------------------------
        loss.backward()

        # ----------------------------------------------------
        # Gradient clipping: prevent exploding gradients
        # Clips gradient norm to maximum value of 1.0
        # ----------------------------------------------------
        torch.nn.utils.clip_grad_norm_(
            model.parameters(),
            1.0
        )

        # ----------------------------------------------------
        # Update model parameters using optimizer
        # ----------------------------------------------------
        optimizer.step()

        # ----------------------------------------------------
        # Update learning rate scheduler
        # ----------------------------------------------------
        scheduler.step()

        # ----------------------------------------------------
        # Accumulate batch loss for epoch average
        # ----------------------------------------------------
        total_loss += loss.item()

    # --------------------------------------------------------
    # Compute average training loss for this epoch
    # --------------------------------------------------------
    train_loss = (
        total_loss /
        len(train_loader)
    )

    # --------------------------------------------------------
    # Run evaluation on validation set
    # Uses threshold=0.35 for recall-sensitive evaluation
    # --------------------------------------------------------
    result = evaluate(
        valid_loader,
        threshold=0.35
    )

    # --------------------------------------------------------
    # Print epoch metrics for monitoring convergence
    # --------------------------------------------------------
    print(f"\nEpoch {epoch+1}")
    print(f"Train Loss : {train_loss:.4f}")
    print(f"Val Loss   : {result.loss:.4f}")
    print(f"Val Acc    : {result.acc:.4f}")
    print(f"Val F1     : {result.f1:.4f}")

    # --------------------------------------------------------
    # Save best model checkpoint if validation loss improved
    # --------------------------------------------------------
    if result.loss < best_val_loss:

        best_val_loss = result.loss
        counter       = 0

        # ----------------------------------------------------
        # Save model state dictionary to disk
        # ----------------------------------------------------
        torch.save(
            model.state_dict(),
            "best_model.pt"
        )

    # --------------------------------------------------------
    # No improvement: increment early stopping counter
    # --------------------------------------------------------
    else:
        counter += 1

    # --------------------------------------------------------
    # Trigger early stopping if patience exceeded
    # --------------------------------------------------------
    if counter >= patience:
        print("\nEarly stopping triggered")
        break

# 17) INFERENCE FUNCTION

# ------------------------------------------------------------
# Predict trafficking probability for a single unseen text
# Used for deployment and real-time analysis
# threshold parameter adjustable for operational requirements
# Lower threshold = higher recall (fewer missed cases)
# Higher threshold = higher precision (fewer false alerts)
# ------------------------------------------------------------
def predict(text, threshold=0.35):

    # --------------------------------------------------------
    # Switch model to evaluation mode
    # --------------------------------------------------------
    model.eval()

    # --------------------------------------------------------
    # Apply same preprocessing as training data
    # --------------------------------------------------------
    cleaned = clean_text(text)

    # --------------------------------------------------------
    # Tokenise cleaned text
    # --------------------------------------------------------
    encoding = tokenizer(

        cleaned,
        truncation=True,
        padding="max_length",
        max_length=MAX_LEN,
        return_tensors="pt"
    )

    # --------------------------------------------------------
    # Move tensors to device
    # --------------------------------------------------------
    encoding = {
        k: v.to(device)
        for k, v in encoding.items()
    }

    # --------------------------------------------------------
    # Disable gradient computation for inference
    # --------------------------------------------------------
    with torch.no_grad():

        # ----------------------------------------------------
        # Forward pass through model
        # ----------------------------------------------------
        logits = model(**encoding).logits

        # ----------------------------------------------------
        # Convert logits to probabilities
        # ----------------------------------------------------
        probs = torch.softmax(
            logits,
            dim=1
        )

        # ----------------------------------------------------
        # Extract trafficking class probability
        # ----------------------------------------------------
        trafficking_prob = probs[0, 1].item()

    # --------------------------------------------------------
    # Apply threshold for final binary prediction
    # --------------------------------------------------------
    prediction = (
        "Trafficking"
        if trafficking_prob >= threshold
        else "Legitimate"
    )

    # --------------------------------------------------------
    # Return structured prediction output
    # --------------------------------------------------------
    return {
        "probability": round(trafficking_prob, 4),
        "prediction":  prediction
    }
